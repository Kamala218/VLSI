module EdgeHighlighter (
    input wire clk,
    input wire rst,
    input wire signal_in,
    output reg rising_pulse,
    output reg falling_pulse
);
    reg prev;

    always @(posedge clk or posedge rst) begin
        if (rst) begin
            rising_pulse <= 0;
            falling_pulse <= 0;
            prev <= 0;
        end else begin
            rising_pulse <= (~prev & signal_in);
            falling_pulse <= (prev & ~signal_in);
            prev <= signal_in;
        end
    end
endmodule
