// graycoder_tb.sv
`timescale 1ns / 1ps

module graycoder_tb;
  reg  [3:0] binary_in = 4'b0000;
  wire [3:0] gray_out;

  // Instantiate DUT
  GrayCoder dut (
    .binary_in(binary_in),
    .gray_out(gray_out)
  );

  // Generate waveform file
  initial begin
    $dumpfile("dump.vcd");
    $dumpvars(0, graycoder_tb);
  end

  // Stimulus sequence
  initial begin
    $display("time | binary_in -> gray_out");
    $monitor("%0t |      %b     ->    %b", $time, binary_in, gray_out);

    // Sweep through all 16 binary values
    for (integer i = 0; i < 16; i = i + 1) begin
      binary_in = i;
      #5;  // allow combinational output to settle
    end

    #10;
    $finish;
  end
endmodule
