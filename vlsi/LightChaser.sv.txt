module LightChaser (
    input wire clk,
    input wire rst,
    input wire enable,
    output reg [7:0] led
);
    always @(posedge clk or posedge rst) begin
        if (rst)
            led <= 8'b00000001;
        else if (enable)
            led <= {led[6:0], led[7]}; // left rotate
    end
endmodule
