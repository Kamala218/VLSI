module EvenOddFSM (
    input wire clk,
    input wire rst,
    input wire [3:0] in_num,
    input wire valid,
    output reg even_flag
);
    always @(posedge clk or posedge rst) begin
        if (rst)
            even_flag <= 1;
        else if (valid)
            even_flag <= ~in_num[0];  // LSB tells even/odd
    end
endmodule
