module BitBalancer (
    input wire [7:0] in_data,
    input wire clk,
    input wire rst,
    output reg [3:0] count
);
    integer i;

    always @(posedge clk or posedge rst) begin
        if (rst) begin
            count <= 0;
        end else begin
            count = 0;
            for (i = 0; i < 8; i = i + 1) begin
                count = count + in_data[i];
            end
        end
    end
endmodule
